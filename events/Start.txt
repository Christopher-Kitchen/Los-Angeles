country_event = {
	id = Start.7
	hidden = yes
	is_triggered_only = yes

	immediate = { #Происходит сразу
		if = { #Если
			limit = { #Одна из стран
				AND = { #Не одна из стран и не воюет с ней
					NOT = { original_tag = RHB }
					NOT = { has_war_with = RHB }
					NOT = { original_tag = KTB }
					NOT = { has_war_with = KTB }
					NOT = { original_tag = FYB }
					NOT = { has_war_with = FYB }
				}
			}
			declare_war_on = { target = RHB type = annex_everything }
			declare_war_on = { target = KTB type = annex_everything }
			declare_war_on = { target = FYB type = annex_everything }
			
		}
		if = { #Если
			limit = { #Одна из стран
				AND = { #Не одна из стран и не воюет с ней
					NOT = { original_tag = LSV }
					NOT = { has_war_with = LSV }
				}
			}
			declare_war_on = { target = LSV type = annex_everything }
		}
		if = { #Если
			limit = { #Одна из стран
				AND = { #Не одна из стран и не воюет с ней
					NOT = { original_tag = GSF }
					NOT = { has_war_with = GSF }
					NOT = { original_tag = SBF }
					NOT = { has_war_with = SBF }
					NOT = { original_tag = VLA }
				}
			}
			declare_war_on = { target = GSF type = annex_everything }
			declare_war_on = { target = SBF type = annex_everything }
		}
        if = { #Если
			limit = { #Одна из стран
				AND = { #Не одна из стран и не воюет с ней
					NOT = { original_tag = VLA }
					NOT = { has_war_with = VLA }
					NOT = { original_tag = GSF }
					NOT = { original_tag = SBF }
				}
			}
			declare_war_on = { target = VLA type = annex_everything }
		}
	}
}
